# 2. Entropy (S) Proxy = Noise to Signal Ratio or Shannon Entropy
# Let's use Kaufman's Efficiency Ratio (ER) to measure noise. 
# ER = Abs(Net Change) / Sum of Abs(Price Changes)
# High ER = Low Entropy (Directed movement), Low ER = High Entropy (Random walk / Chaos)
def calc_er(series, window=10):
    diff = series.diff()
    net_change = abs(series.iloc[-1] - series.iloc[0])
    sum_abs_change = diff.abs().sum()
    if sum_abs_change == 0:
        return 0
    return net_change / sum_abs_change

# Calculate rolling ER and then map to Entropy (1 - ER)

    if i < 10:
        rolling_er.append(np.nan)

        rolling_er.append(calc_er(window_data))
